表操作 ​

sql
1. 查看所有表
show tables;
2. 查看表结构
desc name;
3. 查看表的创建语句
show create table name;

4. 创建表
create table student(
id int(10) primary key not null auto_increment comment '这是注释',
name varchar(20) not null
)
5. 删除表
drop table student;

修改表(增加列)
alter table student add age int;
修改表(修改列)重命名
alter table student change age new_age int;

修改表(修改列)不改变名，修改类型或约束
alter table student modify age tinyint;
删除列
alter table student drop column_name;

-- CRUD
最简单的查询表
select * from student;

插入数据
insert into student values (id, name, age, sex);
insert into student (name, age) values (name, age);

insert into student (name, age) values (name1, age1), (name2, age2);


删除数据
delete from student;
delete from student where id=1;


修改数据
update student set name='new_name' where id=1;
