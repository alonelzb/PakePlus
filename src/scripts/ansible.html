ssh批量认证 ​

bash
#!/bin/bash

# 利用sshpass批量实现基于key验证脚本

password=123456

command -v sshpass &> /dev/null || echo 'install sshpass'; dnf install -y sshpass &> /dev/null

[ -f ~/.ssh/id_rsa ] || ssh-keygen -N ''

ips="
10.1.1.1
10.1.1.2
10.1.1.11
10.1.1.12
"

for i in $ips; do
    echo $i
    { sshpass -p $password ssh-copy-id -o StrictHostKeyChecking=no $i; } &

done
wait

修改内核参数等 ​

yaml
- hosts: cluster
  remote_user: root
  gather_facts: no

  tasks:
    - name: disable firewall
      systemd_service:
        name: firewalld
        enabled: false
        state: stopped

    - name: 修改内核参数等
      shell: 
        cmd: |
          setenforce 0
          sed -i '/^SELINUX=/c SELINUX=disabled' /etc/selinux/config  # 永久
          swapoff -a
          sed -ri 's/.*swap.*/#&/' /etc/fstab
          timedatectl set-timezone Asia/Shanghai
          dnf install -y chrony
          systemctl enable chronyd
          systemctl start chronyd

          cat > /etc/modules-load.d/k8s.conf <<EOF
          hoverlay
          br_netfilter
          EOF

          modprobe overlay
          modprobe br_netfilter

          cat > /etc/sysctl.d/k8s.conf << EOF
          net.ipv4.ip_forward = 1
          net.bridge.bridge-nf-call-ip6tables = 1
          net.bridge.bridge-nf-call-iptables = 1
          EOF

          sysctl -p

安装docker ​

yaml
- name: 安装docker
  hosts: cluster
  gather_facts: false
  vars:
    docker_version: '24.0.0'

  tasks:
    - name: check docker is or not installed
      shell: 
        cmd: docker -v
      register: is_install
      failed_when: is_install.stderr != ''
      ignore_errors: yes
      #
    #- debug: msg=is_install

    - name: 卸载docker
      dnf: 
        name: '{{ item }}'
        loop: [ docker, docker-client, docker-common, docker-engine, docker-client-latest]
        state: absent
      ignore_errors: yes
      when: 
        - is_install.stdout != ''
        - 'docker_version not in is_install.stdout'

          #    - name: 停止docker
          #      systemd_service:
          #        name: docker
          #        state: stopped
          #        ignore_errors: yes
          #        daemon_reload: yes
          #        daemon_reexec: yes

          #- fail: msg=手动退出

    - name: 添加docker仓库
      get_url:
        url: https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo
        dest: /etc/yum.repos.d/docker-ce.repo
        force: no

    - name: 安装docker及其依赖
      dnf: name={{ item }}
      loop:
        - docker-ce-{{ docker_version }}
        - docker-ce-cli-{{ docker_version }}
          #- docker-buildx-plugin
          #- docker-compose-plugin
      when: 'docker_version not in is_install.stdout'

    - name: 添加用户
      user:
        name: docker
        group: docker

    - name: 创建daemon.json
      shell:
        cmd: |
          cat > /etc/docker/daemon.json <<EOF
          {
            "registry-mirrors": ["https://qmaqgl5w.mirror.aliyuncs.com"],
            "insecure-registries": ["harbor.zbluo.com", "10.0.0.30"],
            "exec-opts": ["native.cgroupdriver=systemd"]
          }
          EOF

    - name: 设置开机启动
      systemd_service:
        name: docker
        enabled: true
        state: started

    - name: check docker status
      shell:
        cmd: systemctl is-active docker
      register: status
      #register: status
    - debug: msg={{ status.stdout }}

安装cri-dockerd ​

yaml
- hosts: all
  gather_facts: no
  vars:
    cri: 'cri-dockerd-0.3.11-3.el8.x86_64.rpm'

  tasks:
    - name: 下载cri-dockerd 到本机
      get_url:
        url: 'https://github.com/Mirantis/cri-dockerd/releases/download/v0.3.11/{{ cri }}'
        dest: /opt/{{ cri }}
        force: no

    - name: scp到其它主机
      copy:
        src: /opt/{{ cri }}
        dest: /opt/{{ cri }}

    - name: install cri-docker
      shell:
        cmd: rpm -ivh /opt/{{ cri }}
